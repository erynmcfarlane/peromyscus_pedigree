#SBATCH --job-name=variant_calling

#SBATCH --time=01:00:00

%% START BACK HERE AND RUN AGAIN 0.4
vcftools --gzvcf /home/mcfarlas/projects/def-mcfarlas/peromyscus/vcfs/merged_all_samples.vcf.gz --mac 3 --minQ 20 --max-missing 0.4 --minDP 3 --recode --recode-INFO-all --out peromyscus_mac3_Q20_miss0.4_dp3

vcftools --vcf peromyscus_mac3_Q20_miss0.3_dp3.recode.vcf --missing-indv --out missing_indv

mawk '$5 > 0.8' missing_indv.imiss | cut -f1 > lowDP.indv 

vcftools --vcf peromyscus_mac3_Q20_miss0.3_dp3.recode.vcf \
--remove  lowDP.indv \
--recode \
--recode-INFO-all \
--out peromyscus_mac3_Q20_miss0.3_dp3_ind80

vcftools --vcf peromyscus_mac3_Q20_miss0.3_dp3_ind80.recode.vcf \
    --maf 0.01 \
    --recode \
    --recode-INFO-all \
    --out peromyscus_mac3_Q20_miss0.3_dp3_ind80_maf001
