%% START BACK HERE AND RUN AGAIN 0.4
vcftools --gzvcf /home/mcfarlas/projects/def-mcfarlas/peromyscus/vcfs/merged_all_samples.vcf.gz --mac 3 --minQ 30 --max-missing 0.5 --minDP 3 --recode --recode-INFO-all --out peromyscus_mac3_Q30_miss0.6_dp3
--out peromyscus_mac3_Q30_miss0.5_dp3

mawk '$5 > 0.995' missing_indv.imiss | cut -f1 > lowDP.indv

--remove  lowDP.indv \
--recode \
--recode-INFO-all \
--out peromyscus_mac3_Q30_miss0.6_dp3_ind995

    --maf 0.01 \
    --recode \
    --recode-INFO-all \
    --out peromyscus_mac3_Q30_miss0.6_dp3_ind995_maf001
